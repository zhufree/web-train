map() 把每个元素通过函数传递到当前匹配集合中，生成包含返回值的新的 jQuery 对象。

     $(function(){
      $(document).on('click', 'button', function(event) {
        event.preventDefault();
        var result = $(':checked').map(function() {
          return this.id;
        }).get().join(',');
        console.log(result);
        $('#result').text(result);
      });
    });
